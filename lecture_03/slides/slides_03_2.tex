\begin{frame}[t]{The rank-nullity theorem}
	\vspace{-0.5cm}
	\begin{theorem}
		Let $L: \R^m \to \R^n$ be a linear transformation. Then
	\vspace{-0.2cm}
		$$
		\rank(L) + \dim(\Ker(L)) = m.
	\vspace{-0.2cm}
		$$
	\end{theorem}
\end{frame}
